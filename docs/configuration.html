Wellidate - instance configuration 

// Any input, input container or form.
var form = document.querySelector('form');

new Wellidate(form, {
    // Indicates if first invalid input should be focused after `form` method or submit has failed validation.
    focusInvalid: true,
    // Indicates if first validatable input focus after failed validation should clear the errors.
    focusCleanup: false,
    // Validatable input class when one or more remote validations are pending.
    inputPendingClass: 'input-validation-pending',
    // Validatable input class when at least one of validations failed.
    inputErrorClass: 'input-validation-error',
    // Validatable input class when all validations have passed.
    inputValidClass: 'input-validation-valid',
    // Validatable error container class when one or more remote validations are pending.
    fieldPendingClass: 'input-validation-pending',
    // Validatable error container class when at least one of validations failed.
    fieldErrorClass: 'field-validation-error',
    // Validatable error container class when all validations have passed.
    fieldValidClass: 'field-validation-valid',
    // Selector to match validatable items.
    include: 'input,textarea,select',
    // List of not validatable element selectors.
    excludes: [
        'input[type=button]',
        'input[type=submit]',
        'input[type=image]',
        'input[type=reset]',
        ':disabled'
    ],
    // Summary declaration.
    summary: {
        // Summary container in which results will be rendered.
        container: '[data-valmsg-summary=true]',
        // Validation result callback.
        show: function (result) {
        },
        // Validation reset callback.
        reset: function () {
        }
    },
    // Rules to apply on elements identified by a selector.
    rules: {
        'input': {
            required: { }
        },
        '#username': {
            length: {
                min: 6,
                max: 30
            }
        }
    }
});
