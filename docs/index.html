--- title: Wellidate - Javascript element validator ---
Validation using HTML5 attributes
Using data attributes
Html for validation attributes 

<form>
    <div class="row">
        <div class="col-md-4 mb-3">
            <label for="name">Name</label>
            <input name="name" required maxlength="10">
            <span data-valmsg-for="name"></span>
        </div>
        <div class="col-md-4 mb-3">
            <label for="surname">Surname</label>
            <input name="name" required minlength="2" maxlength="10">
            <span data-valmsg-for="surname"></span>
        </div>
        <div class="col-md-4 mb-3">
            <label for="birthday">Birthday</label>
            <input type="date" name="birthday" required>
            <span data-valmsg-for="birthday"></span>
        </div>
    </div>
</form>

<script>
    new Wellidate(document.querySelector('form'));
</script>
Html for data attribute validation 

<form>
    <div class="row">
        <div class="col-md-4 mb-3">
            <label for="data-name">Name</label>
            <input name="data-name" data-val-required="" data-val-length="" data-val-length-max="10">
            <span data-valmsg-for="data-name"></span>
        </div>
        <div class="col-md-4 mb-3">
            <label for="data-surname">Surname</label>
            <input name="data-surname"  data-val-required="" data-val-length=""
                   data-val-length-min="2" data-val-length-max="10">
            <span data-valmsg-for="data-surname"></span>
        </div>
        <div class="col-md-4 mb-3">
            <label for="data-birthday">Birthday</label>
            <input name="data-birthday" data-val-required="" data-val-date="">
            <span data-valmsg-for="data-birthday"></span>
        </div>
    </div>
</form>

<script>
    new Wellidate(document.querySelector('form'));
</script>
