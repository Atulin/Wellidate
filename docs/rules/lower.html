--- title: Wellidate - lower rule ---
Wellidate - lower rule 



    







HTML options

1) <input name="number" data-val-lower="" data-val-lower-than="10">
2) <input name="number" data-val-lower="Message" data-val-lower-than="10">
3) <input name="number">
4) <input name="number">

<span data-valmsg-for="number"></span>



JS initialization

// Any input, input container or a form.
const container = document.querySelector('form');

1-2) new Wellidate(container);
3) new Wellidate(container, {
    rules: {
        'input': {
            lower: {
                than: 10
            }
        }
    }
  });
4) new Wellidate(container, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        'input': {
            lower: {
                // Required, maximum numeric value which is not allowed to be entered.
                than: 10,
                // Optional, trim value before validating.
                trim: true,
                // Optional, failed validation message.
                message: 'Please enter a value lower than {0}.',
                // Optional, format message in desired format.
                formatMessage() {
                    return this.message.replace('{0}', this.min);
                },
                // Optional, filter enabled when function.
                isEnabled() {
                    return true;
                },
                // Optional, override validation.
                isValid() {
                    const value = this.normalizeValue();

                    return !value || parseFloat(value) < this.than;
                },
                // Optional, override value normalization for the rule.
                normalizeValue() {
                    return this.trim ? this.element.value.trim() : this.element.value;
                }
            }
        }
    }
  });
