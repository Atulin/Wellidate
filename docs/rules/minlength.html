Wellidate - minlength rule 



    







HTML options

1) <input name="username" minlength="2">
2) <input name="username" minlength="2" data-val-minlength="Message">
3) <input name="username" data-val-minlength="" data-val-minlength-min="2">
4) <input name="username" data-val-minlength="Message" data-val-minlength-min="2">
5) <input name="username">
6) <input name="username">

<span data-valmsg-for="username"></span>



JS initialization

// Any input, input container or a form.
const container = document.querySelector('form');

1-4) new Wellidate(container);
5) new Wellidate(container, {
    rules: {
        'input': {
            minlength: {
                min: 2
            }
        }
    }
  });
6) new Wellidate(container, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        'input': {
            minlength: {
                // Required, minimum number of characters that should be entered.
                min: 2,
                // Optional, trim value before validating.
                trim: true,
                // Optional, failed validation message.
                message: 'Please enter at least {0} characters.',
                // Optional, format message in desired format.
                formatMessage() {
                    return this.message.replace('{0}', this.min);
                },
                // Optional, filter enabled when function.
                isEnabled() {
                    return true;
                },
                // Optional, override validation.
                isValid() {
                    return this.min <= this.normalizeValue().length;
                },
                // Optional, override value normalization for the rule.
                normalizeValue() {
                    return this.trim ? this.element.value.trim() : this.element.value;
                }
            }
        }
    }
  });
