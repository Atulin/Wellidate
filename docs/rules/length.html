Wellidate - length rule 



    







HTML options

1) <input name="username" data-val-length="" data-val-length-min="2" data-val-length-max="10">
2) <input name="username" data-val-length="Message" data-val-length-min="2" data-val-length-max="10">
3) <input name="username">
4) <input name="username">

<span data-valmsg-for="username"></span>



JS initialization

// Any input, input container or a form.
const container = document.querySelector('form');

1-2) new Wellidate(container);
3) new Wellidate(container, {
    rules: {
        'input': {
            length: {
                min: 2,
                max: 10
            }
        }
    }
  });
4) new Wellidate(container, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        'input': {
            length: {
                // Semi required, minimum number of characters that should be entered, can be ommited if max value is specified.
                min: 2,
                // Semi required, maximum number of characters that can be entered, can be ommited if min value is specified.
                max: 10,
                // Optional, trim value before validating.
                trim: true,
                // Optional, failed validation message.
                message: 'Please enter a value between {0} and {1} characters long.',
                // Optional, format message in desired format.
                formatMessage() {
                    return this.message.replace('{0}', this.min).replace('{1}', this.max);
                },
                // Optional, filter enabled when function.
                isEnabled() {
                    return true;
                },
                // Optional, override validation.
                isValid() {
                    const value = this.normalizeValue();

                    return (this.min == null || this.min <= value.length) && (value.length <= this.max || this.max == null);
                },
                // Optional, override value normalization for the rule.
                normalizeValue() {
                    return this.trim ? this.element.value.trim() : this.element.value;
                }
            }
        }
    }
  });
