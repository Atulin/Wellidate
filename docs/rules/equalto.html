Wellidate - equalto rule 



    







HTML options

<input id="password">

1) <input name="password-repeat" data-val-equalto="" data-val-equalto-other="password">
2) <input name="password-repeat" data-val-equalto="Message" data-val-equalto-other="password">
3) <input name="password-repeat">
4) <input name="password-repeat">

<span data-valmsg-for="password-repeat"></span>



JS initialization

// Any input, input container or a form.
const container = document.querySelector("form");

1-2) new Wellidate(container);
3) new Wellidate(container, {
    rules: {
        "[name=password-repeat]": {
            equalto: {
                other: "password"
            }
        }
    }
  });
4) new Wellidate(container, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        "[name=password-repeat]": {
            required: {
                // Optional, trim value before validating.
                trim: true,
                // Required, other element id to compare with.
                other: "password",
                // Optional, failed validation message.
                message: "Please enter the same value again.",
                // Optional, format message in desired format.
                formatMessage() {
                    return this.message;
                },
                // Optional, filter enabled when function.
                isEnabled() {
                    return true;
                },
                // Optional, override validation.
                isValid() {
                    const other = document.getElementById(this.other);

                    return other != null && this.normalizeValue(this.element) == this.normalizeValue(other);
                },
                // Optional, override value normalization for the rule.
                normalizeValue(element) {
                    return this.trim ? element.value.trim() : element.value;
                }
            }
        }
    }
  });
