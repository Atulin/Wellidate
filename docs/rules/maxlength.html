Wellidate - maxlength rule 



    
    

        
    







HTML options

1) <input name="username" maxlength="10">
2) <input name="username" maxlength="10" data-val-maxlength="Message">
3) <input name="username" data-val-maxlength="" data-val-maxlength-max="10">
4) <input name="username" data-val-maxlength="Message" data-val-maxlength-max="10">
5) <input name="username">
6) <input name="username">

<span data-valmsg-for="username"></span>



JS initialization

// Any input, input container or form.
var form = document.querySelector('form');

1-4) new Wellidate(form);
5) new Wellidate(form, {
    rules: {
        'input': {
            maxlength: {
                max: 10
            }
        }
    }
  });
6) new Wellidate(form, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        'input': {
            maxlength: {
                // Required, maximum number of characters that should be entered.
                max: 10,
                // Optional, trim value before validating.
                trim: true,
                // Optional, failed validation message.
                message: 'Please enter no more than {0} characters.',
                // Optional, format message in desired format.
                formatMessage: function () {
                    return this.message.replace('{0}', this.max);
                },
                // Optional, filter enabled when function.
                isEnabled: function () {
                    return true;
                },
                // Optional, override validation.
                isValid: function () {
                    return this.normalizeValue().length <= this.max;
                },
                // Optional, override value normalization for the rule.
                normalizeValue: function () {
                    return this.trim ? this.element.value.trim() : this.element.value;
                }
            }
        }
    }
  });
