Wellidate - step rule 



    







HTML options

1) <input name="number" step="3">
2) <input name="number" step="3" data-val-step="Message">
3) <input name="number" data-val-step="" data-val-step-value="3">
4) <input name="number" data-val-step="Message" data-val-step-value="3">
5) <input name="number">
6) <input name="number">

<span data-valmsg-for="number"></span>



JS initialization

// Any input, input container or form.
var form = document.querySelector('form');

1-4) new Wellidate(form);
5) new Wellidate(form, {
    rules: {
        'input': {
            step: {
                value: 3
            }
        }
    }
  });
6) new Wellidate(form, {
    rules: {
        // Required, any valid document selector for which to apply the rules.
        'input': {
            lower: {
                // Required, numeric value divisor for which input value has to divide without a remainder.
                value: 3,
                // Optional, trim value before validating.
                trim: true,
                // Optional, failed validation message.
                message: 'Please enter a multiple of {0}.',
                // Optional, format message in desired format.
                formatMessage: function () {
                    return this.message.replace('{0}', this.value);
                },
                // Optional, filter enabled when function.
                isEnabled: function () {
                    return true;
                },
                // Optional, override validation.
                isValid: function () {
                    var value = this.normalizeValue();

                    return !value || parseFloat(value) % this.value == 0;
                },
                // Optional, override value normalization for the rule.
                normalizeValue: function () {
                    return this.trim ? this.element.value.trim() : this.element.value;
                }
            }
        }
    }
  });
